465   class VerboseDownloadReporter
 
466     attr_reader :file_name, :total_bytes, :progress
 
467 
468     def initialize(out_stream, *args)

474       @file_name = file_name
 
475       @total_bytes = total_bytes.to_i
 
476       @units = @total_bytes.zero? ? 'B' : '%'

481     def update(bytes)

483                        bytes
 
484                      else
 
485                        ((bytes.to_f * 100) / total_bytes.to_f).ceil
 
486                      end
 
487
