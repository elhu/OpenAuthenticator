334   def shebang(bin_file_name)
 
335     ruby_name = Gem::ConfigMap[:ruby_install_name] if @env_shebang
 
336     path = File.join @gem_dir, @spec.bindir, bin_file_name
 
337     first_line = File.open(path, "rb") {|file| file.gets}
 
338 
339     if /\A#!/ =~ first_line then
 
340       # Preserve extra words on shebang line, like "-w".  Thanks RPA.
 
341       shebang = first_line.sub(/\A\#!.*?ruby\S*(?=(\s+\S+))/, "#!#{Gem.ruby}")
 
342       opts = $1
 
343       shebang.strip! # Avoid nasty ^M issues.
 
344     end
 
345 
346     if not ruby_name then
 
347       "#!#{Gem.ruby}#{opts}"
 
348     elsif opts then
 
349       "#!/bin/sh\n'exec' #{ruby_name.dump} '-x' \"$0\" \"$@\"\n#{shebang}"
 
350     else
 
351       # Create a plain shebang line.
 
352       @env_path ||= ENV_PATHS.find {|env_path| File.executable? env_path }
 
353       "#!#{@env_path} #{ruby_name}"
 
354     end

476   def build_extensions
 
477     return if @spec.extensions.empty?
 
478     say "Building native extensions.  This could take a while..."
 
479     dest_path = File.join @gem_dir, @spec.require_paths.first
 
480     ran_rake = false # only run rake once
 
481 
482     @spec.extensions.each do |extension|
 
483       break if ran_rake
 
484       results = []
 
485

487                 when /extconf/ then
 
488                   Gem::Ext::ExtConfBuilder
 
489                 when /configure/ then
 
490                   Gem::Ext::ConfigureBuilder
 
491                 when /rakefile/i, /mkrf_conf/i then
 
492                   ran_rake = true
 
493                   Gem::Ext::RakeBuilder
 
494                 else
 
495                   results = ["No builder for extension '#{extension}'"]
 
496                   nil
 
497                 end
 
498 
499 
500       extension_dir = begin
 
501                         File.join @gem_dir, File.dirname(extension)
 
502                       rescue TypeError # extension == nil
 
503                         @gem_dir
 
504                       end
 
505 
506 
507       begin
 
508         Dir.chdir extension_dir do

510 
511           say results.join("\n") if Gem.configuration.really_verbose
 
512         end
 
513       rescue
 
514         results = results.join "\n"
 
515 
516         gem_make_out = File.join extension_dir, 'gem_make.out'
 
517

521 ERROR: Failed to build gem native extension.
 
522

524 
525 Gem files will remain installed in #{@gem_dir} for inspection.

539   def extract_files
 
540     @gem_dir = File.expand_path @gem_dir
 
541 
542     raise ArgumentError, "format required to extract from" if @format.nil?
 
543 
544     dirs = []
 
545 
546     @format.file_entries.each do |entry, file_data|
 
547       path = entry['path'].untaint
 
548 
549       if path =~ /\A\// then # for extra sanity
 
550         raise Gem::InstallError,
 
551               "attempt to install file into #{entry['path'].inspect}"
 
552       end
 
553 
554       path = File.expand_path File.join(@gem_dir, path)
 
555 
556       if path !~ /\A#{Regexp.escape @gem_dir}/ then
 
557         msg = "attempt to install file into %p under %p" %
 
558                 [entry['path'], @gem_dir]
 
559         raise Gem::InstallError, msg
 
560       end
 
561 
562       FileUtils.rm_rf(path) if File.exists?(path)
 
563 
564       dir = File.dirname(path)
 
565       if !dirs.include?(dir)
 
566         dirs << dir
 
567         FileUtils.mkdir_p dir
 
568       end
 
569 
570       File.open(path, "wb") do |out|
 
571         out.write file_data
 
572       end
 
573 
574       FileUtils.chmod entry['mode'], path
 
575 
576       say path if Gem.configuration.really_verbose
 
577     end
