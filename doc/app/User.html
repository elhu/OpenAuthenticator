def self.new_with_params params
  traits = params[:user]
  user = User.new(traits)
  user.login = traits[:login]
  date = traits[:birthdate]
  begin
    user.birthdate = Date.new(date[:year].to_s.to_i, date[:month].to_s.to_i, date[:day].to_s.to_i)
  rescue
    user.birthdate = nil
  end
  user
end
